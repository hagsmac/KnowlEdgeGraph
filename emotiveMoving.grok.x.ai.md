import pandas as pd

def calculate_reversion_score(game: dict, sport: str) -> float:
    db = pd.read_csv("data/reversion_db.csv")
    historical = db[(db["sport"] == sport) & (db["team"] == game["fave"])]["win_rate"].mean() or 0.50

    weights = {
        "historical": 0.45,
        "comeback_rank": 0.25,
        "deficit_factor": 0.15,      # 0-1 goal = 1.0, halftime 5pt = 0.9
        "motivation": 0.10,          # playoff chase = 1.2
        "steam_intensity": 0.05
    }

    score = (
        historical * weights["historical"] +
        game.get("comeback_rank", 0.5) * weights["comeback_rank"] +
        game.get("deficit_factor", 0.8) * weights["deficit_factor"] +
        game.get("motivation_factor", 1.0) * weights["motivation"] +
        game.get("steam_intensity", 1.0) * weights["steam_intensity"]
    )
    return round(min(score, 0.99), 3)
